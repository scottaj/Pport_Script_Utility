require ‘rubygems’ require ‘rake’ require ‘rake/clean’ require ‘rake/gempackagetask’ require ‘rake/rdoctask’ require ‘rake/testtask‘

spec = Gem::Specification.new do |s| 

  s.name = 'fworks_script_utility'
  s.version = '0.4.1'
  s.has_rdoc = true
  s.extra_rdoc_files = ['README', 'LICENSE']
  s.summary = 'A utility for scripting fireworks or other light shows'
  s.description = s.summary
  s.author = 'Al Scott'
  s.email = 'scottaj2@udmercy.edu'
  # s.executables = ['your_executable_here']
  s.files = %w(LICENSE README Rakefile) + Dir.glob("{bin,lib,spec}/**/*")
  s.require_path = "lib"
  s.bindir = "bin"
  s.add_runtime_dependency 'wxruby'
  s.add_runtime_dependency 'wx_sugar'
  s.add_runtime_dependency 'Pport'

end

Rake::GemPackageTask.new(spec) do |p| 

  p.gem_spec = spec
  p.need_tar = true
  p.need_zip = true

end

Rake::RDocTask.new do |rdoc| 

  files =['README', 'LICENSE', 'lib/**/*.rb', 'lib/**/*.rbw']
  rdoc.external = true
  rdoc.rdoc_files.add(files)
  rdoc.main = "README" # page to start on
  rdoc.title = "fworks_script_utility Docs"
  rdoc.rdoc_dir = 'doc/rdoc' # rdoc output folder
  rdoc.options << '--line-numbers'
  rdoc.options << '--all'
  rdoc.options << '--promiscuous'

end

Rake::TestTask.new do |t| 

  t.test_files = FileList['test/**/*.rb']

end
